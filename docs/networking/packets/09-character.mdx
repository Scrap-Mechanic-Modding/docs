title: Character
---

Contains all information about a character.

- **ID**: 0x09
- **Size**: Variable
- **State**: Handshaking
- **Bound To**: Bi-directional

## Structure

```
//------------------------------------------------
//--- 010 Editor v12.0 Binary Template
//
//      File: SM_Packet_9
//   Authors: ColdMeekly
//   Version: 1.0.0
//   Purpose: Yes
//  Category: 
// File Mask: 
//  ID Bytes: 09 FA 7F
//   History: 
//------------------------------------------------

typedef struct {
    byte data[16];
} uuid;

typedef struct {
  
  byte id;
  char magic[2];
  byte Hair; // A3 = leftmost hair, A7 = rightmost hair
  byte name_len;
  char name[name_len]; // TODO: Check how name affects packet
  char magic_char[4];
  // Check if magic is 00 00 00 02
  byte is_male;
  byte num_uuids;
  uuid uuids[num_uuids];
  BigEndian();
  uint32 skin_id; // 0-4
  LittleEndian();
  char magic2[11];
} SMPacket9 <optimize=false>;


FSeek(0);
```
